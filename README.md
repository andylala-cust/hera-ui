> A Vue.js UI Kit for Web.

## Install
```shell
npm install hera-ui -S
```

## Quick Start
``` javascript
import Vue from 'vue'
import HeraUi from 'hera-ui'

Vue.use(HeraUi)

// or
import {
  icon,
  panel
  // ...
} from 'hera-ui'

Vue.component(icon.name, icon)
Vue.component(panel.name, panel)
```
